 create table conductor (id_conductor int  not null auto_increment, nombre varchar (30) not null, primary key (id_conductor));
 create table programa (id_programa int  not null auto_increment, nombre_programa varchar (20) not null, activo boolean not null, primary key (id_programa));
 create table conduce (id_conductor int  not null, id_programa int not null, fecha_inicio date not null, actividad_conductor boolean not null, primary key (id_programa, id_conductor), foreign key (id_conductor) references conductor (id_conductor), foreign key (id_programa) references programa (id_programa));
 create table alquiler (id_alquiler int not null auto_increment, couta int (7) unsigned not null, primary key (id_alquiler));
 create table tiene (fecha_couta date not null,pago boolean not null, id_alquiler int not null, id_programa int not null, primary key (id_alquiler, id_programa), foreign key (id_alquiler) references alquiler (id_alquiler), foreign key (id_programa) references programa (id_programa));
 create table fecha_programa (id_fecha_programa  int not null auto_increment, fecha_inicio date not null, fecha_final date not null, primary key (id_fecha_programa));
 create table emite (id_programa int not null, id_fecha_programa int not null, activo_en_horario boolean not null, primary key (id_programa, id_fecha_programa), foreign key (id_fecha_programa) references fecha_programa (id_fecha_programa), foreign key (id_programa) references programa (id_programa));
 create table empresa (id_empresa int not null auto_increment, nombre varchar (20) not null, primary key (id_empresa));
 create table publicidad (id_publicidad int  not null auto_increment,tema varchar (60) not null, id_empresa int not null, primary key (id_publicidad), foreign key (id_empresa) references empresa (id_empresa));
 create table tanda (id_tanda int not null auto_increment, primary key (id_tanda));
 create table horas (id_tanda int not null, horas time not null, primary key (id_tanda), foreign key (id_tanda) references tanda (id_tanda));
 create table aparece (id_publicidad int  not null,activa boolean not null,pago boolean not null, id_tanda int  not null, primary key (id_publicidad, id_tanda), foreign key (id_tanda) references tanda(id_tanda), foreign key (id_publicidad) references publicidad (id_publicidad));
 create table serie (id_serie int  not null auto_increment, nombre varchar (30) not null, activo bool not null, primary key (id_serie));
 create table video (id_video int  not null auto_increment, contenido varchar (40) not null, nombre varchar (30) not null,emitido boolean not null,id_serie int not null, fecha date not null, primary key (id_video), foreign key (id_serie) references serie (id_serie));
